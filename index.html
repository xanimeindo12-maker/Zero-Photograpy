// Script Code.gs untuk Backend Website Foto (Versi Keamanan OTP Gmail) function doPost(e) { if (typeof e !== 'undefined' && e.postData && e.postData.contents) { var data = JSON.parse(e.postData.contents); if (data.action === "beli_foto") { return catatPesanan(data); } else if (data.action === "request_otp") { return kirimOTP(); } else if (data.action === "upload_foto") { return uploadFotoKeDrive(data); } } return ContentService.createTextOutput("Error: Permintaan Tidak Valid"); } // ================= FUNGSI 1: KIRIM OTP KE GMAIL ================= function kirimOTP() { // Sistem otomatis mendeteksi email Google yang Anda gunakan untuk membuat script ini var adminEmail = Session.getEffectiveUser().getEmail(); // Membuat 6 digit angka acak var otpCode = Math.floor(100000 + Math.random() * 900000).toString(); // Menyimpan OTP di memori sementara (Cache) selama 1 jam (3600 detik) var cache = CacheService.getScriptCache(); cache.put("ADMIN_OTP", otpCode, 3600); // Mengirim Email var subject = "🔒 Kode Akses Dashboard Fotoyu"; var body = "Halo Admin,\n\n" + "Seseorang sedang mencoba masuk ke Dashboard Fotoyu.\n" + "Jika ini Anda, silakan gunakan kode rahasia berikut:\n\n" + "Kode Akses: " + otpCode + "\n\n" + "Kode ini hanya berlaku selama 1 jam.\n\n" + "Sistem Keamanan Fotoyu."; MailApp.sendEmail(adminEmail, subject, body); return ContentService.createTextOutput(JSON.stringify({ "status": "success", "pesan": "Kode OTP telah dikirim ke email Anda!" })).setMimeType(ContentService.MimeType.JSON); } // ================= FUNGSI 2: UPLOAD FOTO (Validasi OTP) ================= function uploadFotoKeDrive(data) { var cache = CacheService.getScriptCache(); var validOtp = cache.get("ADMIN_OTP"); // Validasi Keamanan Lapis Baja if (!validOtp || validOtp !== data.otp) { return ContentService.createTextOutput(JSON.stringify({ "status": "error", "pesan": "Akses Ditolak: Kode OTP Salah atau Sudah Kedaluwarsa!" })).setMimeType(ContentService.MimeType.JSON); } try { // PASTIKAN ID FOLDER INI SUDAH DIISI DENGAN ID MILIK ANDA var FOLDER_ID = "MASUKKAN_ID_FOLDER_ANDA_DISINI"; var folder = DriveApp.getFolderById(FOLDER_ID); var contentType = data.mimeType; var byteCharacters = Utilities.base64Decode(data.fileData); var blob = Utilities.newBlob(byteCharacters, contentType, data.nama_file); folder.createFile(blob); return ContentService.createTextOutput(JSON.stringify({ "status": "success", "pesan": "Foto " + data.nama_file + " berhasil diunggah!" })).setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ "status": "error", "pesan": "Gagal upload: " + error.message })).setMimeType(ContentService.MimeType.JSON); } } // ================= FUNGSI 3: BELI FOTO (Tetap Sama) ================= function catatPesanan(data) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Sheet1"); var waktu = new Date(); var idTransaksi = "TRX-" + Math.floor(100000 + Math.random() * 900000); var idFoto = data.id_foto; var nominal = data.harga; var status = "Pending"; var FOLDER_ID = "1YNZNmu1ISKdmFQt-SN8SCdhM5c0Pq1NQ"; // Isi ID Folder Anda var linkDownload = "File tidak ditemukan"; try { var folder = DriveApp.getFolderById(FOLDER_ID); var files = folder.searchFiles('title contains "' + idFoto + '"'); if (files.hasNext()) { linkDownload = files.next().getUrl(); } } catch(error) { linkDownload = "Error Akses Drive"; } sheet.appendRow([waktu, idTransaksi, idFoto, nominal, status, linkDownload]); var response = { "status": "success", "id_transaksi": idTransaksi, "link_bayar": "https://simulator.sandbox.midtrans.com/" }; return ContentService.createTextOutput(JSON.stringify(response)).setMimeType(ContentService.MimeType.JSON); }
